Criar tabela de discursantes
```sql
create table speakers (
	first_speaker varchar(255),
  first_theme varchar(255),
  second_speaker varchar(255),
  second_theme varchar(255),
  third_speaker varchar(255),
  third_theme varchar(255),
  created_at date,
  created_by date,
  updated_at date,
  updated_by date
)
```
